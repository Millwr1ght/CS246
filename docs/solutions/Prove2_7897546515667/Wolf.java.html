1    package macbeth;
2    
3    import java.awt.*;
4    import java.util.*;
5    
6    /**
7     * Defines a Wolf which is a creature that can move, be aggresive, is aware around him, and can
8     * spawn new wolves.
9     *
10    * @author Chad Macbeth (CS246 Prove-02-Stretch)
11    */
12   public class Wolf extends Creature implements Movable, Aggressor, Aware, Spawner {
13       private Direction direction;
14       private Boolean canSpawn;
15   
16       /**
17        * Create a wolf with a randomly set direction.
18        */
19       public Wolf() {
20           Random random = new Random();
21           int randomDirection = random.nextInt(4);
22   
23           if (randomDirection == 0) {
24               direction = Direction.Up;
25           }
26           else if (randomDirection == 1) {
27               direction = Direction.Down;
28           }
29           else if (randomDirection == 2) {
30               direction = Direction.Left;
31           }
32           else {
33               direction = Direction.Right;
34           }
35   
36           _health = 1;
37           canSpawn = false;  // Can only spawn after killing prey
38       }
39   
40       /**
41        * Wolves are square
42        *
43        * @return {@link Shape} of the wolf
44        */
45       public Shape getShape() {
46           return Shape.Square;
47       }
48   
49       /**
50        * Wolves are grey
51        *
52        * @return {@link Color} of the wolf
53        */
54       public Color getColor() {
55           return Color.gray;
56       }
57   
58       /**
59        * Get health status of the wolfe.
60        * Wolves are alive only when their health is positive.
61        *
62        * @return alive
63        */
64       public Boolean isAlive() {
65           return (_health > 0);
66       }
67   
68       /**
69        * Wolf will move one square at a time based on their
70        * current preferred direction.
71        */
72       public void move() {
73           if (direction == Direction.Up) {
74               _location.y--;
75           }
76           else if (direction == Direction.Right) {
77               _location.x++;
78           }
79           else if (direction == Direction.Down) {
80               _location.y++;
81           }
82           else if (direction == Direction.Left) {
83               _location.x--;
84           }
85           // If invalid direction then do not move.
86       }
87   
88       /**
89        * Update the preferred direction of the wolf based on the creatures around
90        * the wolf.  Will first check in their current preferred direction.  If no Animal
91        * is found, then the wolf will check clockwise for an animal.  If still no Animal
92        * is found, then the preferred direction is not changed.  If an Animal is found, then
93        * the preferred direction is changed towards that Animal.
94        *
95        * @param above The {@link Creature} directly above us.
96        * @param below The {@link Creature} directly below us.
97        * @param left The {@link Creature} directly to the left of us.
98        * @param right The {@link Creature} directly to the right of us.
99        */
100      public void senseNeighbors(Creature above, Creature below, Creature left, Creature right) {
101          if (direction == Direction.Up) {
102              if (above instanceof Animal) {
103                  direction = Direction.Up;
104              }
105              else if (right instanceof Animal) {
106                  direction = Direction.Right;
107              }
108              else if (below instanceof Animal) {
109                  direction = Direction.Down;
110              }
111              else if (left instanceof Animal) {
112                  direction = Direction.Left;
113              }
114          }
115          else if (direction == Direction.Right) {
116              if (right instanceof Animal) {
117                  direction = Direction.Right;
118              }
119              else if (below instanceof Animal) {
120                  direction = Direction.Down;
121              }
122              else if (left instanceof Animal) {
123                  direction = Direction.Left;
124              }
125              else if (above instanceof Animal) {
126                  direction = Direction.Up;
127              }
128          }
129          else if (direction == Direction.Down) {
130              if (below instanceof Animal) {
131                  direction = Direction.Down;
132              }
133              else if (left instanceof Animal) {
134                  direction = Direction.Left;
135              }
136              else if (above instanceof Animal) {
137                  direction = Direction.Up;
138              }
139              else if (right instanceof Animal) {
140                  direction = Direction.Right;
141              }
142          }
143          else if (direction == Direction.Left) {
144              if (left instanceof Animal) {
145                  direction = Direction.Left;
146              }
147              else if (above instanceof Animal) {
148                  direction = Direction.Up;
149              }
150              else if (right instanceof Animal) {
151                  direction = Direction.Right;
152              }
153              else if (below instanceof Animal) {
154                  direction = Direction.Down;
155              }
156          }
157      }
158  
159      /**
160       * Inflict 5 points of health damage if the creature is an Animal.
161       * Also allow the wolf to spawn one time.
162       *
163       * @param target The {@link Creature} we've encounterd.
164       */
165      public void attack(Creature target) {
166          if(target instanceof Animal) {
167              target.takeDamage(5);
168              canSpawn = true;
169          }
170      }
171  
172      /**
173       * If the wolf can spawn, then create a new wolf to the left.
174       * If the wolf can not spawn, then null is returned.
175       *
176       * @return new {@link Creature} spawned
177       */
178      public Creature spawnNewCreature() {
179          if (canSpawn) {
180              canSpawn = false;
181              Wolf spawn = new Wolf();
182              spawn.setLocation(new Point(_location.x - 1, _location.y));
183              return spawn;
184          }
185          return null;
186      }
187  }
