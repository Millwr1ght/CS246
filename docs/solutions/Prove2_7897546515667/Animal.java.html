1    package macbeth;
2    
3    import java.awt.*;
4    import java.util.Random;
5    
6    /**
7    *  Animals move around and eat plants. They are represented by red squares.
8    * <p>
9    * @author  Brother Falin
10   * @version 1.0
11   * @since   2016-12-08 
12   * @see Creature
13   */
14   public class Animal extends Creature implements Movable, Aggressor, Spawner {
15   
16   	Random _rand;
17   	private boolean canSpawn;
18   	private boolean isSleeping;
19   	private int sleepTimer;
20   
21   	/**
22   	 * Creates an animal with 1 health point.
23   	 */
24   	public Animal() {
25   		_rand = new Random();
26   		_health = 1;
27   		canSpawn = false;
28   		isSleeping = false;
29   		sleepTimer = 0;
30   	}
31   
32   	// No javadocs are necessary for these methods because they will inherit the
33   	// documentation from the superclass. We only need to add docs here if we are
34   	// doing something non-obvious in our overridden version.
35   
36   	public Boolean isAlive() {
37   		return _health > 0;
38   	}
39   
40   	public Shape getShape() {
41   		return Shape.Square;
42   	}
43   
44   	public Color getColor() {
45   		if (isSleeping) {
46   			return Color.black;
47   		}
48   		return new Color(165, 33, 135);
49   	}
50   
51   	/**
52   	 * If the creature we've encountered is a plant, we'll eat it. Otherwise, we ignore it.
53   	 * @param target The {@link Creature} we've encountered.
54   	 */
55   	public void attack(Creature target) {
56   		// Animals only eat plants. Give the plant 1 damage
57   		// and increase our health by one.
58   		if (isSleeping)
59   		{
60   			return;
61   		}
62   		if(target instanceof Plant) {
63   			target.takeDamage(1);
64   			_health++;
65   			canSpawn = true;
66   		}
67   	}
68   
69   	/**
70   	 * Move the animal in a random direction.
71   	 */
72   	public void move() {
73   
74   		// Choose a random direction each time move() is called.
75   		if (isSleeping)
76   		{
77   			return;
78   		}
79   		switch(_rand.nextInt(4)) {
80   			case 0:
81   				_location.x++;
82   				break;
83   			case 1:
84   				_location.x--;
85   				break;
86   			case 2:
87   				_location.y++;
88   				break;
89   			case 3:
90   				_location.y--;
91   				break;
92   			default:
93   				break;
94   		}
95   	}
96   
97   	// Spawn a new 20% of the time after eating
98   	public Creature spawnNewCreature() {
99   		if (_rand.nextInt(4) == 0 && canSpawn) {
100  			Animal spawn = new Animal();
101  			spawn.setLocation(new Point(_location.x - 1, _location.y));
102  			canSpawn = false;
103  			return spawn;
104  		}
105  		return null;
106  	}
107  
108  }
