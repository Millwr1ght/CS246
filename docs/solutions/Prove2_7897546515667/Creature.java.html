Creature.java 
1    package macbeth;
2    
3    import java.awt.Color;
4    import java.awt.Point;
5    
6    // Since this is an abstract class, it cannot be created using new.
7    // Instead, we have to create subclasses of this class.
8    
9    /**
10   *  Creature is the base class for all creatures in the game. However, unless they implement
11   *  one of the defined interfaces, they just sit there and do nothing.
12   * <p>
13   * @author  Brother Falin
14   * @version 1.0
15   * @since   2016-12-08 
16   */
17   public abstract class Creature {
18   	
19   	Point _location;
20   	int _health;
21   	
22   	/**
23   	* Gets the location of the creature in game coordinates.
24   	* @return The current location of the creature.
25   	*/
26   	public Point getLocation() {
27   		return _location;
28   	}
29   	
30   	/**
31   	* Sets the location of the creature in game coordinates.
32   	* @param newValue The new location of the creature. 
33   	*/
34   	public void setLocation(Point newValue) {
35   		_location = newValue;
36   	}
37   	
38   	/**
39   	* Allows the creature to take damage if something comes by to attack it.
40   	* @param damage The amount of damage inflicted by the attacker
41   	*/
42   	public void takeDamage(int damage) {
43   		_health -= damage;
44   	}
45   	
46   	// Since the following methods are marked as abstract, subclasses
47   	// must implement them. 
48   
49   	/**
50   	* Gets the {@link Shape} used to represent this creature.
51   	* @return The shape drawn for this creature.
52   	*/
53   	abstract Shape getShape();
54   
55   	/**
56   	* Gets the {@link java.awt.Color} used to represent this creature.
57   	* @return The color drawn for this creature.
58   	*/
59   	abstract Color getColor();
60   
61   	/**
62   	* Is the creature alive?
63   	* @return A boolean indicating if the creature is alive.
64   	*/
65   	abstract Boolean isAlive();
66   }
