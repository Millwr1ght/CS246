9    /**
10   * The "brains" of the game, which controls all of the creature activites.
11   * <p>
12   * @author  Brother Falin
13   * @version 1.0
14   * @since   2016-12-08 
15   * @see Creature
16   */
17   public class CreatureHandler
18   {
19   	int _rows;
20   	int _cols;
21   	Random _rand;
22   	List<Creature> _creatures; 
23   
24   	/**
25   	* Retrieves all of the creatures that exist in the world
26   	* @return A list of the creatures in the world.
27   	*/
28   	public List<Creature> getCreatures() {
29   		return _creatures;
30   	}
31   
32   	/**
33   	* @param worldRows How many rows the world grid should have.
34   	* @param worldCols How many columns the world grid should have.
35   	* @param creatures A list of creatures to start the world with.
36   	*/
37   	public CreatureHandler(int worldRows, int worldCols, List<Creature> creatures) {
38   		_rand = new Random();
39   		_rows = worldRows;
40   		_cols = worldCols;
41   		_creatures = creatures;
42   	}
43   
44   	/**
45   	* Sets the initial positions of the creatures to random locations within the 
46   	* world's grid.
47   	*/
48   	public void setStartingPositions() {
49   
50   		// Java's enhanced for loop (AKA for-each loop).
51   		// See: http://stackoverflow.com/a/11685345/28106
52   		for(Creature c : _creatures) {
53   			c.setLocation(getRandomLocation());
54   		}
55   	}
56   
57   	private Point getRandomLocation() {
58   		int r = _rand.nextInt(_rows);
59   		int c = _rand.nextInt(_cols);
60   		
61   		return new Point(r, c);
62   	}
63   
64   	private void wrapPoint(Point p) {
65   
66   		// If we go too far left or right, wrap around
67   		// to the other side.
68   		if(p.x > _cols)
69   			p.x = 0;
70   		else if(p.x < 0)
71   			p.x = _cols;
72   		
73   		// If we go too far up or down, wrap around
74   		// to the other side.
75   		if(p.y > _rows)
76   			p.y = 0;
77   		else if(p.y < 0)
78   			p.y = _rows;
79   	}
80   
81   	private Creature getTarget(Creature source, int xOffset, int yOffset) {
82   		
83   		// Determine where we are looking based on our current location
84   		// and the offset
85   		Point newPoint = (Point)source.getLocation().clone();
86   		newPoint.x += xOffset;
87   		newPoint.y += yOffset;
88   
89   		// Wrap the target point if necessary
90   		wrapPoint(newPoint);
91   
92   		// Look at every creature in our list, and see if it is sitting in the spot
93   		// where we are looking. (ignoring dead creatures)
94   		for(Creature c : _creatures) {
95   			if(c == source || !c.isAlive())
96   				continue;
97   				
98   			if(c.getLocation().x == newPoint.x && 
99   			   c.getLocation().y == newPoint.y)
100  				return c;
101  		}
102  		
103  		// If we didn't find anything, return null.
104  		return null;
105  	}
106  
107  	private void handleMove(Creature c) {
108  		
109  		Movable m = (Movable)c;
110  		m.move();
111  		
112  		wrapPoint(c.getLocation());
113  	}
114  
115  	/**
116  	* Updates all of the living creatures in the game by making they each get a chance to
117  	* act on their implemented behaviors.
118  	*/
119  	public void updateCreatures() {
120  
121  		// Handle all our creature behaviors here. Since we don't know ahead of time
122  		// which creatures implement which behaviors, we can use the instanceof keyword
123  		// to see if a given instance implements a particular interface.
124  		List<Creature> newCreatures = new ArrayList<Creature>();
125  
126  		for(Creature c : _creatures) {
127  				
128  				// Skip dead creatures
129  				if(!c.isAlive())
130  					continue;
131  
132  				if(c instanceof Aware) {
133  					Creature above = getTarget(c, 0, -1);
134  					Creature below = getTarget(c, 0, 1);
135  					Creature left = getTarget(c, -1, 0);
136  					Creature right = getTarget(c, 1, 0);
137  
138  					Aware a = (Aware)c;
139  					a.senseNeighbors(above, below, left, right);
140  				}
141  
142  				if(c instanceof Movable) {
143  					handleMove(c);
144  				}
145  				
146  				if(c instanceof Aggressor) {
147  					Creature target = getTarget(c, 0, 0);
148  					Aggressor a = (Aggressor)c;
149  					a.attack(target);
150  				}
151  
152  				// Prompt all creatures that implement the Spawner
153  				// interface to spawn a new creature.
154  				if (c instanceof Spawner) {
155  					Spawner spawner = (Spawner) c;
156  					Creature newCreature = spawner.spawnNewCreature();
157  					if (newCreature != null)
158  					{
159  						// If a new creature was created, then add it to
160  						// a list of new creatures.  We don't want to modify
161  						// the _creatures list since we are currently
162  						// iterating through it ... the iterator would
163  						// be adversely disrupted.
164  						newCreatures.add(newCreature);
165  					}
166  				}
167  
168  				// Prompt all creature that implement the Grower
169  				// interface to plant.
170  				if (c instanceof Grower) {
171  					Grower grower = (Grower) c;
172  					Creature newCreature = grower.plant();
173  					if (newCreature != null)
174  					{
175  						// As with the spawner above, need to add
176  						// any new creatures (plants) to the 
177  						// new creatures list.
178  						newCreatures.add(newCreature);
179  					}
180  				}
181  			}
182  			
183  			// Now that the iterator is finished, add any new creatures
184  		    // that were created earlier.
185  			_creatures.addAll(newCreatures);
186  	}
