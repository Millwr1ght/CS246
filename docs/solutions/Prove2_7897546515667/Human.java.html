1    package macbeth;
2    
3    import java.awt.*;
4    import java.util.Random;
5    
6    /**
7     * Defines a Human which is a creature that is aggresive, is aware of its surroundings, can move
8     * can spawn, and can grow things.
9     *
10    * @author Chad Macbeth (CS246 Prove-02-Stretch)
11    */
12   public class Human extends Creature implements Aggressor, Aware, Movable, Spawner, Grower {
13       private Direction direction;
14       private Random random;
15   
16       /**
17        * Create a Human moving up with initial health.
18        */
19       public Human() {
20   
21           random = new Random();
22   
23           direction = Direction.Up;
24   
25           _health = 10; // Initial Health
26       }
27   
28       /**
29        * Humans are circles.
30        *
31        * @return {$Link Shape} of humans
32        */
33       public Shape getShape() {
34           return Shape.Circle;
35       }
36   
37       /**
38        * Humans are orange.
39        *
40        * @return {$Link Color} of humans.
41        */
42       public Color getColor() {
43           return Color.orange;
44       }
45   
46       /**
47        * Get health status of the human.
48        * Humans are only alive when their health is positive.
49        *
50        * @return alive
51        */
52       public Boolean isAlive() {
53           return (_health > 0);
54       }
55   
56       /**
57        * Human will move one square at a time based on their
58        * current preferred direction.
59        */
60       public void move() {
61           if (direction == Direction.Up) {
62               _location.y--;
63           } else if (direction == Direction.Right) {
64               _location.x++;
65           } else if (direction == Direction.Down) {
66               _location.y++;
67           } else if (direction == Direction.Left) {
68               _location.x--;
69           }
70       }
71   
72       /**
73        * If an animal or wolf is sensed, then the human will move in that direction.
74        * If nothing is found, then the human will randomly select a direction to search.
75        *
76        * @param above The {@link Creature} directly above us.
77        * @param below The {@link Creature} directly below us.
78        * @param left The {@link Creature} directly to the left of us.
79        * @param right The {@link Creature} directly to the right of us.
80        */
81       public void senseNeighbors(Creature above, Creature below, Creature left, Creature right) {
82           if (above instanceof Animal || above instanceof Wolf) {
83               direction = Direction.Up;
84           } else if (right instanceof Animal || right instanceof Wolf) {
85               direction = Direction.Right;
86           } else if (below instanceof Animal || below instanceof Wolf) {
87               direction = Direction.Down;
88           } else if (left instanceof Animal || left instanceof Wolf) {
89               direction = Direction.Left;
90           } else {
91               switch (random.nextInt(4)) {
92                   case 0:
93                       _location.x++;
94                       break;
95                   case 1:
96                       _location.x--;
97                       break;
98                   case 2:
99                       _location.y++;
100                      break;
101                  case 3:
102                      _location.y--;
103                      break;
104                  default:
105                      break;
106              }
107          }
108      }
109  
110      /**
111       * Attempt to kill any animal or wolf that is found and Increase health by 10 points.
112       * Killing an animal or wolf will only work 50% of the time.
113       * Failure to kill will result in loss of 10 points
114       *
115       * Kill completely any plant that is found and increase health by 2 point.
116       *
117       * @param target The {@link Creature} we've encounterd.
118       */
119      public void attack(Creature target) {
120          if (target instanceof Animal || target instanceof Wolf) {
121              if (random.nextInt(2) != 0) {
122                  target.takeDamage(target._health);
123                  _health += 10;
124              } else {
125                  _health -= 10;
126              }
127          }
128          if (target instanceof Plant) {
129              target.takeDamage(target._health);
130              _health += 2;
131          }
132      }
133  
134      /**
135       * If the health of the human > 40, then spawn a new human 20% of the time.
136       * The new baby will be to the left.  The health of the human will go back to 10.
137       * If no new human, then will return null.
138       *
139       * @return new {$Link Creature}
140       */
141      public Creature spawnNewCreature() {
142          if (random.nextInt(5) == 0 && _health > 20) {
143              Human spawn = new Human();
144              spawn.setLocation(new Point(_location.x - 1, _location.y));
145              _health = 10;
146              return spawn;
147          }
148          return null;
149      }
150  
151      /**
152       * The human will successfully plant 10% of the time.
153       * The new plant will be to the left.
154       * If no new plant, then will return null.
155       *
156       * @return new {$Link Creature}
157       */
158      public Creature plant() {
159          if (random.nextInt(10) == 0)
160          {
161              Plant plant = new Plant();
162              plant.setLocation(new Point(_location.x - 1, _location.y));
163              return plant;
164          }
165          return null;
166      }
167  }
