1    package macbeth;
2    
3    import java.awt.Canvas;
4    import java.awt.Graphics;
5    import java.awt.Color;
6    import javax.swing.JFrame;
7    import java.util.List;
8    import java.util.ArrayList;
9    import java.awt.Point;
10   import java.awt.Font;
11   import java.awt.Dimension;
12   import java.awt.event.ComponentAdapter;
13   import java.awt.event.ComponentEvent;
14   
15   /**
16   * Handles drawing and the passage of time in the world.
17   * <p>
18   * @author  Brother Falin
19   * @version 1.0
20   * @since   2016-12-08 
21   * @see CreatureHandler
22   */
23   public class World extends Canvas implements Runnable  {
24   	
25   	Dimension _world;
26   	Dimension _cell;
27   	int _totalStartingCreatures;
28   	
29   	CreatureHandler _handler;
30   	
31   	/**
32   	* Initializes the world, the drawing window, and starts the update thread in the background.
33       * @param creatures The {@link Creature} instances that exist at the beginning of the game.
34       */
35   	public World(List<Creature> creatures) {
36   
37   		_totalStartingCreatures = creatures.size();
38   		
39   		// You can adjust the overall dimensions of the world by modifying this line
40   		_world = new Dimension(400, 400);
41   
42   		// The size of each cell depends on how many creatures we have and the
43   		// size of the world.
44   		_cell = new Dimension(_world.width / _totalStartingCreatures, 
45   		                      _world.height / _totalStartingCreatures);
46   		
47   		// The CreatureHandler class keeps track of where the creatures are,
48   		// but it needs to know the size of the world.
49   		int rows = _world.height / _cell.height;
50   		int cols = _world.width / _cell.width;
51   		_handler = new CreatureHandler(rows, cols, creatures);
52   		_handler.setStartingPositions();
53   
54   		// This is all AWT stuff for setting up the window
55   		JFrame win = new JFrame("Life with Java");
56   		win.setSize(_world);
57   		win.setDefaultCloseOperation(JFrame.EXIT_ON_CLOSE);
58   		win.getContentPane().setBackground(new Color(233, 232, 231));
59   		win.add( this );
60   		
61   		// This strange little construction is called an anonymous inner class.
62   		// Here, we're using it to handle window resizing because that's how AWT
63   		// makes us do it.
64   		// You can read more about anonymous classes here: 
65   		// https://docs.oracle.com/javase/tutorial/java/javaOO/anonymousclasses.html
66   		win.addComponentListener(new ComponentAdapter() {
67   			@Override
68   			public void componentResized(ComponentEvent e)
69   			{
70   				_world = new Dimension(getWidth(), getHeight());
71   				_cell = new Dimension(_world.width / _totalStartingCreatures, 
72   				                      _world.height / _totalStartingCreatures);
73   				repaint();
74   			}
75   		});
76   
77   		win.setVisible(true);
78   		
79   		// Since this class implements the Runnable interface, we can 
80   		// have the simulation updating happen on a background thread.
81   		Thread t = new Thread(this);
82   		t.start();
83   	}
84   	
85   	/**
86   	* This function is called on the background thread by virtue of the {@link java.lang.Runnable}
87   	* interface.
88       * <p>
89       * Every time it's called, we update all of the creatures, redraw the window, and take a brief .25
90       * sleep so the game doesn't run blindingly fast.
91       */
92   	public void run() {
93   		
94   		// This infinite loop will run as long as the simulation is
95   		// active.
96   		while(true)
97   		{
98   			// Give all of the creatures a chance to respond to their
99   			// enviroment.
100  			_handler.updateCreatures();
101  
102  			// Then redraw everything in its new position
103  			repaint();
104  			
105  			// Pause the background thread for 0.25 seconds so we can watch
106  			// the simulation unfold. Note how we have to wrap the sleep()
107  			// call in a try/catch block.
108  			try {
109  				Thread.sleep(50);
110  			} catch (Exception e) {
111  				
112  			}
113  			
114  		}
115  	}
116  
117  	// This function will inherit its docs from the Canvas class, so we'll just
118  	// add some inline developer notes.
119  	public void paint( Graphics context ) {
120  
121  		// Java's enhanced for loop (AKA for-each loop).
122  		// See: http://stackoverflow.com/a/11685345/28106
123  		for(Creature c : _handler.getCreatures()) {
124  			
125  			// Don't draw things that are dead
126  			if(!c.isAlive())
127  				continue;
128  			
129  			Point p = c.getLocation();
130  			int x = p.x * _cell.width;
131  			int y = p.y * _cell.height;
132  			
133  			context.setColor(c.getColor());
134  			
135  			// You can add additional creature visualizations by adding 
136  			// additional values to the Shape enum (in Shape.java)
137  			// and handling them here. 
138  			// For details on what kind of drawing you can do, see:
139  			// https://docs.oracle.com/javase/7/docs/api/java/awt/Graphics.html
140  			switch (c.getShape()) {
141  				case Circle:
142  					context.fillOval(x, y, _cell.width, _cell.height);
143  					break;
144  				case Square:
145  					context.fillRect(x, y, _cell.width, _cell.height);
146  					break;
147  				default:
148  					break;
149  			}
150  		}
151  		
152  		// Draw the grid lines
153  		context.setColor(new Color(187, 187, 185));
154  		int rows = _world.height / _cell.height;
155  		int cols = _world.width / _cell.width;
156  
157  		for(int r = 0; r < rows; r++) {
158  			context.drawLine(0, r * _cell.height, _world.width, r * _cell.height);
159  		}
160  		
161  		for(int c = 0; c < cols; c++) {
162  			context.drawLine(c * _cell.width, 0, c * _cell.width, _world.height);
163  		}
164  		
165  	}
166  }
