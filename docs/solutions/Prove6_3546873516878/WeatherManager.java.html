3    import com.google.gson.Gson;

5    public class WeatherManager {
6        private final String key = "ADD API KEY HERE";
7        private final String baseUrl = "https://api.openweathermap.org/data/2.5/";
8        private Gson gson;
9        private HTTPHelper httpHelper;
10   
11       public WeatherManager() {
12           gson = new Gson();
13           httpHelper = new HTTPHelper();
14       }
15   
16       public WeatherConditions loadCurrentWeather(String city) {
17           // Request for weather data
18           String url = baseUrl + "weather" +
19                   "?q=" + city +
20                   "&units=imperial" +
21                   "&apiKey=" + key;
22           String data = httpHelper.readHTTP(url);
23   
24           // Convert JSON data to WeatherConditions object
25           return gson.fromJson(data, WeatherConditions.class);
26       }
27   
28       public WeatherForecast loadWeatherForecast(String city) {
29           // Requset for weather forecast
30           String url = baseUrl + "forecast" +
31                   "?q=" + city +
32                   "&units=imperial" +
33                   "&apiKey=" + key;
34           String data = httpHelper.readHTTP(url);
35   
36           // Convert JSON data to WeatherForecast object
37           return gson.fromJson(data, WeatherForecast.class);
38       }
