1    package t.macbeth.prove05;
2    
3    import androidx.appcompat.app.AppCompatActivity;
4    
5    import android.content.Context;
6    import android.content.Intent;
7    import android.content.SharedPreferences;
8    import android.os.Bundle;
9    import android.util.Log;
10   import android.view.View;
11   import android.widget.EditText;
12   
13   public class PromptScripture extends AppCompatActivity {
14   
15       private EditText bookId;
16       private EditText chapterId;
17       private EditText verseId;
18   
19       /**
20        * Start the Activity
21        *
22        * @param savedInstanceState
23        */
24       @Override
25       protected void onCreate(Bundle savedInstanceState) {
26           super.onCreate(savedInstanceState);
27           setContentView(R.layout.activity_prompt_scripture);
28   
29           bookId = (EditText) findViewById(R.id.etBook);
30           chapterId = (EditText) findViewById(R.id.etChapter);
31           verseId = (EditText) findViewById(R.id.etVerse);
32       }
33   
34       /**
35        * Respond to submit button.
36        *
37        * @param view
38        */
39       public void submitFavoriteScripture(View view) {
40           // Create Intent for next Activity
41           Intent intent = new Intent(this, DisplayScripture.class);
42   
43           String book = bookId.getText().toString();
44           String chapter = chapterId.getText().toString();
45           String verse = verseId.getText().toString();
46   
47           intent.putExtra("BOOK", book);
48           intent.putExtra("CHAPTER", chapter);
49           intent.putExtra("VERSE", verse);
50   
51           Log.d(this.getLocalClassName(),"About to create intent with " +
52                   book + " " + chapter + ":" + verse + ".");
53   
54           // Start the next activity
55           startActivity(intent);
56   
57       }
58   
59       /**
60        * Load the Scripture from Shared Preferences
61        *
62        * @param view
63        */
64       public void loadScripture(View view) {
65           // Get a shared preferences handle for the application
66           SharedPreferences sharedPref = getSharedPreferences(
67                   "Scripture.txt", Context.MODE_PRIVATE);
68   
69           // Get the configuration values
70           String book = sharedPref.getString("BOOK", "");
71           String chapter = sharedPref.getString("CHAPTER", "");
72           String verse = sharedPref.getString("VERSE", "");
73   
74           // Populate the Text Fields
75           bookId.setText(book);
76           chapterId.setText(chapter);
77           verseId.setText(verse);
78   
79       }
80   }
