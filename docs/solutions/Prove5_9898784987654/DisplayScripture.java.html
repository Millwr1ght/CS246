1    package t.macbeth.prove05;
2    
3    import androidx.appcompat.app.AppCompatActivity;
4    
5    import android.content.Context;
6    import android.content.Intent;
7    import android.content.SharedPreferences;
8    import android.os.Bundle;
9    import android.util.Log;
10   import android.view.View;
11   import android.widget.TextView;
12   import android.widget.Toast;
13   
14   public class DisplayScripture extends AppCompatActivity {
15   
16       private String book;
17       private String chapter;
18       private String verse;
19   
20       /**
21        * Start the Activity
22        * Display the Favorite Scripture in Normal Format
23        *
24        * @param savedInstanceState
25        */
26       @Override
27       protected void onCreate(Bundle savedInstanceState) {
28           super.onCreate(savedInstanceState);
29           setContentView(R.layout.activity_display_scripture);
30   
31           // Populate the fields based on the Intent Extra's
32           Intent intent = getIntent();
33           book = intent.getStringExtra("BOOK");
34           chapter = intent.getStringExtra("CHAPTER");
35           verse = intent.getStringExtra("VERSE");
36   
37           String scripture = book + " " + chapter + ":" + verse;
38           Log.d(this.getLocalClassName(),"Received intent with " + scripture);
39           TextView scriptureObject = findViewById(R.id.tvScripture);
40           scriptureObject.setText(scripture);
41       }
42   
43       /**
44        * Save the Scripture to Shared Preferences
45        *
46        * @param view
47        */
48       public void saveScripture(View view) {
49           // Get a shared preferences handle for the application
50           SharedPreferences sharedPref = getSharedPreferences(
51                   "Scripture.txt", Context.MODE_PRIVATE);
52           SharedPreferences.Editor editor = sharedPref.edit();
53   
54           // Set the configuration values and commit
55           editor.putString("BOOK", book);
56           editor.putString("CHAPTER", chapter);
57           editor.putString("VERSE", verse);
58           editor.apply();
59   
60           // Display Toast to User
61           Toast.makeText(this, "Scripture Saved", Toast.LENGTH_SHORT).show();
62       }
63   }
