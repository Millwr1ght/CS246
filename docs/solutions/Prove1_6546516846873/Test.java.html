8     * @author Chad Macbeth (CS246 Prove-01-Stretch)

10   public class Test {
11   
12       /**
13        * Prompt for a password and then hash the password.
14        * Test to ensure the algorithm worked.
15        *
16        * @param args - Command Line Arguments
17        */
18       public static void main(String[] args) {
19           Scanner scanner = new Scanner(System.in);
20           System.out.print("Enter password: ");
21           String password = scanner.nextLine();
22           User user = new User(password);
23           try {
24               // Only Password should be set at this point
25               System.out.println("Password: " + user.getPassword());
26               System.out.println("Salt: " + user.getSalt());
27               System.out.println("Hashed Password: " + user.getHashedPassword());
28   
29               // hashUserPassword is a static function (no NSALoginController object needed)
30               // This function will set the salt and hashedPassword in the user object
31               // We expect that the password will be cleared.
32               NSALoginController.hashUserPassword(user);
33   
34               // Only salt and hashed password set at this point
35               System.out.println("Password: " + user.getPassword());
36               System.out.println("Salt: " + user.getSalt());
37               System.out.println("Hashed Password: " + user.getHashedPassword());
38   
39               // Run a test of the hashed password.
40               // Ask the user for the password and set it in the user object
41               System.out.print("Enter password: ");
42               password = scanner.nextLine();
43               user.setPassword(password);
44   
45               // verifyPassword will check the password in the user object against the salt
46               // and hashed password.
47               if (NSALoginController.verifyPassword(user)) {
48                   System.out.println("Verified!");
49               }
50               else {
51                   System.out.println("Not Verified!");
52               }
53           }
54           catch (WeakPasswordException e) {
55               System.out.println("Weak Password Error: " + e.getMessage());
56           }
57           catch(Exception e) {
58               System.out.println("General Exception: " + e.toString());
59           }
60   
61       }
62   }
