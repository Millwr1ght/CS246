65   public final class NSALoginController {
66   	
67   	// The SecureRandom() class is a special subclass of Random() in 
68   	// order to make our "random" salt values less predictable.
69   	// For more information, see:
70   	// http://stackoverflow.com/questions/11051205/difference-between-java-util-random-and-java-security-securerandom
71   	private static final Random RANDOM = new SecureRandom();
72   
73   	// The use of "static final" here is the equivalent of "const" in C++
74   	// These constant values are used by the hash algorithm.
75   	private static final int ITERATIONS = 10000;
76   	private static final int KEY_LENGTH = 256;
77   
78   	// By using a private constructor, we prevent instances of this class from being created
79   	private NSALoginController() {
80   
81   	}
82   
83   	// This function generates a random, 16-byte 
84   	// salt value. You might be wondering if a longer salt
85   	// would result in a more secure hash:
86   	// http://stackoverflow.com/questions/184112/what-is-the-optimal-length-for-user-password-salt
87   	private static byte[] getNextSalt() {
88   		byte[] salt = new byte[16];
89   		RANDOM.nextBytes(salt);
90   		return salt;
91   	}
92   	
93   	// This function uses the PBKDF2 algorithm for generating the hash. If you're 
94   	// interested in why this particular function was chosen, see:
95   	// http://security.stackexchange.com/questions/4781/do-any-security-experts-recommend-bcrypt-for-password-storage/6415#6415
96   	private static byte[] getHash(char[] password, byte[] salt) throws Exception {
97   
98   		PBEKeySpec spec = new PBEKeySpec(password, salt, ITERATIONS, KEY_LENGTH);
99   		Arrays.fill(password, Character.MIN_VALUE);
100  
101  		SecretKeyFactory skf = SecretKeyFactory.getInstance("PBKDF2WithHmacSHA1");
102  		byte[] hash = skf.generateSecret(spec).getEncoded();
103  	
104  		return hash;
105  	}
106  	
107  	/**
108  	* This function takes the password from the {@link User} class and hashes it.
109  	* As a side-effect, the original password value is removed for security purposes.
110  	* @param user The user whose password needs to be hashed.
111  	* @exception Exception If there is a problem with the chosen hash function.
112  	* @exception WeakPasswordException If there is a weak password detected.
113  	*/
114  	public static void hashUserPassword(User user) throws Exception {
115  		
116  		// Get the next random salt value to use for this password
117  		byte[] salt = getNextSalt();
118  		char[] password = user.getPassword().toCharArray();
119  
120  		// Verify password rules:
121  		// 1) Length at least 8 characters
122  		// 2) Contain at least one digit
123  		if (password.length < 8)
124  		{
125  			throw new WeakPasswordException("Password must be at least 8 characters long.");
126  		}
127  		boolean hasDigit = false;
128  		for (int i=0; i<password.length; i++)
129  		{
130  			if (Character.isDigit(password[i]))
131  			{
132  				hasDigit = true;
133  			}
134  		}
135  		if (!hasDigit)
136  		{
137  			throw new WeakPasswordException("Password must contain at least 1 digit.");
138  		}
139  		// Once we've generated the hash, clear the old password
140  		// from memory for security purposes
141  		byte[] hash = getHash(password, salt);
142  		Arrays.fill(password, Character.MIN_VALUE);
143  		user.setPassword("");
144  		
145  		if(hash != null) {
146  
147  			// By Base64-encoding the raw bytes, we can store them as strings.
148  			// This allows us to save the values to a file or database if needed.
149  			// For more information on Base64 encoding, see:
150  			// http://stackoverflow.com/questions/201479/what-is-base-64-encoding-used-for
151  			// https://en.wikipedia.org/wiki/Base64
152  			String hashString = Base64.getEncoder().encodeToString(hash);
153  			String saltString = Base64.getEncoder().encodeToString(salt);
154  			
155  			user.setHashedPassword(hashString);
156  			user.setSalt(saltString);
157  		}
158  		else
159  		{
160  			user.setHashedPassword(null);
161  			user.setSalt(null);
162  		}
163  	}
164  	
165  	/**
166  	* This function uses the password and salt in the {@link User} to generate a hash,
167  	* then compares that hash to the original hash value.
168  	* @param user The user whose password needs to be hashed.
169  	* @return Whether or not the password values match.
170  	* @exception Exception If there is a problem with the chosen hash function.
171  	*/
172  	public static Boolean verifyPassword(User user) throws Exception {
173  		
174  		// Have to get the raw data values to use on our hash function
175  		char[] password = user.getPassword().toCharArray();
176  		byte[] salt = Base64.getDecoder().decode(user.getSalt());
177  		
178  		// Generate the new hash, and retrieve the user's hash
179  		byte[] expectedHash = getHash(password, salt);
180  		byte[] userHash = Base64.getDecoder().decode(user.getHashedPassword());
181  		
182  		// If the new hash came out as null, or the lengths don't match,
183  		// we know that the original password is different
184  		if(expectedHash == null || expectedHash.length != userHash.length)
185  			return false;
186  		
187  		// Check each byte of the two hashes and as soon as we find one
188  		// that is different, we know that the passwords don't match
189  		for(int i = 0; i < expectedHash.length; i++) {
190  			if(expectedHash[i] != userHash[i])
191  				return false;
192  		}
193  		
194  		// If we got this far, it means the password hashes match, so we
195  		// can assume the passwords do as well.
196  		return true;
197  	}
